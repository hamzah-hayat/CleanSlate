##########################################################################
# Clean Slate Auto-Build Events
# Fires monthly to direct idle construction ships to build starbases
# and hyper relays based on the active policy.
##########################################################################

namespace = cs_auto_build

# Event 1: Auto-Build Starbases
# Fires monthly for countries with the auto-build starbase policy active.
# Finds an unclaimed system and directs an idle construction ship to claim it,
# provided the empire can afford the influence cost.
# Scope: country
country_event = {
	id = cs_auto_build.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ai = no
		OR = {
			has_policy_flag = auto_build_starbases_adjacent
			has_policy_flag = auto_build_starbases_all
		}
		# Must have at least one idle construction ship
		any_owned_ship = {
			is_ship_class = shipclass_constructor
			is_fleet_idle = yes
		}
		# Must be able to afford an outpost
		resource_stockpile_compare = {
			resource = influence
			value >= 75
		}
	}

	immediate = {
		if = {
			limit = { has_policy_flag = auto_build_starbases_adjacent }
			# Systems one jump from an existing starbase, not yet claimed
			random_system_within_border = {
				limit = {
					is_owned_by = root
					NOT = { any_starbase_in_system = { always = yes } }
					any_neighbor_system = {
						is_owned_by = root
						any_starbase_in_system = { always = yes }
					}
				}
				save_event_target_as = cs_starbase_target_system
			}
		}
		else_if = {
			limit = { has_policy_flag = auto_build_starbases_all }
			random_system_within_border = {
				limit = {
					is_owned_by = root
					NOT = { any_starbase_in_system = { always = yes } }
				}
				save_event_target_as = cs_starbase_target_system
			}
		}

		if = {
			limit = { exists = event_target:cs_starbase_target_system }
			event_target:cs_starbase_target_system = {
				create_starbase = {
					owner = root
					size = starbase_outpost
				}
			}
		}
	}
}
