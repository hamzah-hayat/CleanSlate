decision_clear_world = {
	icon = decision_burning_settlement
	sound = event_structural_collapse

	enactment_time = 30

	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}

	potential = {
		owner = {
			is_ai = no
		}
	}

	owned_planets_only = yes

	effect = {
		planet = {
			# Remove all buildings
			remove_all_buildings = yes
			# Remove all districts
			remove_all_districts = yes
			# Remove all zone specialisations from all districts
			every_district = {
				remove_zone = {
					district = this
					all = yes
				}
			}

			# Make sure to re-add the capital building
			hidden_effect = {
				# We have the following capitals we can re-add
				# Habitat capitals - Habitats
				# Standard capitals - Non-Hive Empires
				# Hive capitals - Hive Empires
				# Machine Capitals - Machine Empires
				if = {
					limit = {
						is_planet_class = pc_habitat
					}
					add_building = building_hab_capital
				}
				else_if = {
					limit = { 
						owner = {
							is_regular_empire = yes
						}
					}
					add_building = building_colony_shelter
				}
				else_if = {
					limit = { 
						owner = {
							is_hive_empire = yes
						}
					}
					add_building = building_hive_capital
				}
				else_if = {
					limit = { 
						owner = {
							is_wilderness_empire = yes
						}
					}
					add_building = building_colony_shelter_wilderness
				}
				else_if = {
					limit = { 
						owner = {
							is_machine_empire = yes
						}
					}
					add_building = building_deployment_post
				}

				# Just in case
				validate_planet_buildings_and_districts = yes
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

decision_expel_population = {
	icon = decision_resettle_pops
	sound = event_administrative

	enactment_time = 30

	resources = {
		category = decisions
		cost = {
			influence = 100
		}
	}

	potential = {
		owner = {
			is_ai = no
		}
		# Must have pops to expel
		num_pops > 0
	}

	owned_planets_only = yes

	allow = {
		custom_tooltip = {
			fail_text = "decision_expel_population_no_destination"
			owner = {
				# Either there are other planets in the same sector, or there are other sectors with planets
				OR = {
					# Other planets in the same sector
					any_owned_planet = {
						NOT = { is_same_value = prev.prev }
						sector = { is_same_value = prev.prev.sector }
						num_pops > 0
					}
					# Other sectors with planets
					any_owned_sector = {
						NOT = { is_same_value = prev.prev.sector }
						any_owned_planet = {
							num_pops > 0
						}
					}
				}
			}
		}
	}

	effect = {
		planet = {
			# Check if there are other planets in the same sector
			if = {
				limit = {
					owner = {
						any_owned_planet = {
							NOT = { is_same_value = prev.prev }
							sector = { is_same_value = prev.prev.sector }
						}
					}
				}
				# Resettle all pop groups to other planets in the same sector
				every_owned_pop_group = {
					save_event_target_as = cs_expel_pop_group
					owner = {
						random_owned_planet = {
							limit = {
								NOT = { is_same_value = prev.prev }
								sector = { is_same_value = prev.prev.sector }
							}
							resettle_pop_group = {
								POP_GROUP = event_target:cs_expel_pop_group
								PLANET = this
							}
						}
					}
				}
			}
			else = {
				# No other planets in this sector — send to a random other sector
				every_owned_pop_group = {
					save_event_target_as = cs_expel_pop_group
					owner = {
						random_owned_sector = {
							limit = {
								NOT = { is_same_value = prev.prev.sector }
								any_owned_planet = {
									always = yes
								}
							}
							random_owned_planet = {
								resettle_pop_group = {
									POP_GROUP = event_target:cs_expel_pop_group
									PLANET = this
								}
							}
						}
					}
				}
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

# Toggle decision: Rapid Construction Mode
# Increases building speed at the cost of increased building costs.
# Scales with city engineering tech level. Cannot be taken while buildings are under construction.
decision_rapid_construction_enable = {
	icon = decision_industrial_development
	sound = event_administrative

	enactment_time = 0

	resources = {
		category = decisions
		cost = {
			unity = 50
		}
	}

	potential = {
		owner = { is_ai = no }
		NOT = { has_planet_flag = cs_rapid_construction_active }
	}

	owned_planets_only = yes

	allow = {
		# Cannot activate while any building is under construction
		custom_tooltip = {
			fail_text = "decision_rapid_construction_no_buildings_under_construction"
			NOT = { has_building_construction = yes }
		}
	}

	effect = {
		set_planet_flag = cs_rapid_construction_active
		# Apply the modifier based on tech level
		if = {
			limit = {
				owner = { has_technology = tech_construction_templates }
			}
			# Tier 5: 1000% speed, 1000% cost
			add_modifier = {
				modifier = cs_rapid_construction_t5
				days = -1
			}
		}
		else_if = {
			limit = {
				owner = { has_technology = tech_assembly_pattern }
			}
			# Tier 4: 700% speed, 700% cost
			add_modifier = {
				modifier = cs_rapid_construction_t4
				days = -1
			}
		}
		else_if = {
			limit = {
				owner = { has_technology = tech_construction_1 }
			}
			# Tier 3: 400% speed, 400% cost
			add_modifier = {
				modifier = cs_rapid_construction_t3
				days = -1
			}
		}
		else_if = {
			limit = {
				owner = { has_technology = tech_basic_industry }
			}
			# Tier 2: 300% speed, 300% cost
			add_modifier = {
				modifier = cs_rapid_construction_t2
				days = -1
			}
		}
		else = {
			# Tier 1: 200% speed, 200% cost
			add_modifier = {
				modifier = cs_rapid_construction_t1
				days = -1
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

# Decision: Destroy Habitat
# Permanently destroys an empty habitat, refunding a small portion of alloys.
decision_destroy_habitat = {
	icon = decision_burning_settlement
	sound = event_structural_collapse

	enactment_time = 30

	resources = {
		category = decisions
		cost = {
			unity = 100
		}
	}

	potential = {
		owner = { is_ai = no }
		is_planet_class = pc_habitat
	}

	owned_planets_only = yes

	allow = {
		# Habitat must be empty
		custom_tooltip = {
			fail_text = "decision_destroy_habitat_must_be_empty"
			num_pops = 0
		}
	}

	effect = {
		# Refund a small portion of alloys
		owner = {
			add_resource = { alloys = 200 }
		}
		# Destroy the habitat megastructure
		solar_system = {
			random_megastructure = {
				limit = {
					is_megastructure_type = habitat_3
				}
				remove_megastructure = this
			}
			random_megastructure = {
				limit = {
					is_megastructure_type = habitat_restored
				}
				remove_megastructure = this
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}
