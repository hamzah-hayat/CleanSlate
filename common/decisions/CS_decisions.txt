# Decision: Clear World
# Removes all buildings and districts from the planet, then re-adds the appropriate
# capital building for the owner's empire type. Useful for resetting a planet to a blank slate.
cs_decision_clear_world = {
	icon = decision_burning_settlement
	sound = event_structural_collapse

	enactment_time = 30

	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}

	potential = {
		owner = {
			is_ai = no
		}
	}

	owned_planets_only = yes

	effect = {
		planet = {
			# Remove all buildings and districts
			remove_all_buildings = yes
			remove_all_districts = yes

			# Remove the district specialization as well from the main district (eg city)
			hidden_effect = {
				remove_zone = {
					district = 0
					zone_slot = 1
				}
				remove_zone = {
					district = 0
					zone_slot = 2
				}
			}

			# Re-add the appropriate capital building
			hidden_effect = {
				if = {
					limit = { is_planet_class = pc_habitat }
					add_building = building_hab_capital
				}
				else_if = {
					limit = { owner = { is_regular_empire = yes } }
					add_building = building_colony_shelter
				}
				else_if = {
					limit = { owner = { is_hive_empire = yes } }
					add_building = building_hive_capital
				}
				else_if = {
					limit = { owner = { is_wilderness_empire = yes } }
					add_building = building_colony_shelter_wilderness
				}
				else_if = {
					limit = { owner = { is_machine_empire = yes } }
					add_building = building_deployment_post
				}
				validate_planet_buildings_and_districts = yes
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

# Decision: Demolish Habitat
# Permanently destroys an empty habitat, refunding a small portion of alloys.
# The habitat must have no population before demolition can begin.
cs_decision_destroy_habitat = {
	icon = decision_burning_settlement
	sound = event_structural_collapse

	enactment_time = 30

	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}

	potential = {
		owner = { is_ai = no }
		is_planet_class = pc_habitat
	}

	owned_planets_only = yes

	allow = {
		custom_tooltip = {
			fail_text = "cs_decision_destroy_habitat_must_be_empty"
			is_colony = no
		}
	}

	effect = {
		# Refund alloys based on shipset type
		# This is by default a 33% refund
		owner = {
			if = {
				limit = {
					country_uses_bio_ships = yes
				}
				add_resource = { alloys = 250 }
			}
			else = {
				add_resource = { alloys = 500 }
			}
		}

		# Remove all habitat megastructures in this system
		solar_system = {
			every_megastructure = {
				limit = {
					OR = {
						is_megastructure_type = habitat_central_complex
						is_megastructure_type = habitat_central_complex_restored
						is_megastructure_type = habitat_major_orbital
						is_megastructure_type = habitat_minor_orbital
					}
				}
				remove_megastructure = this
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

# Decision: Relocate Population
# Resettles all pops on this planet to other planets in the same sector.
# If this is the only planet in the sector, sends pops to a random other sector.
cs_decision_relocate_population = {
	icon = decision_expel_population
	sound = event_administrative_work

	enactment_time = 30

	resources = {
		category = decisions
		cost = {
			influence = 25
			unity = 500
		}
	}

	potential = {
		owner = { is_ai = no }
		any_owned_pop_group = { always = yes }
	}

	owned_planets_only = yes

	allow = {
		custom_tooltip = {
			fail_text = "cs_decision_relocate_population_no_destination"
			owner = {
				count_owned_sector = { count > 1 }
			}
		}
	}

	effect = {
		custom_tooltip = cs_decision_relocate_population_effect_tooltip
		hidden_effect = {
			planet = {
				# Save planet as event target for comparison
				save_event_target_as = cs_expel_source_planet
				# Save this planet's sector for comparison
				sector = { save_event_target_as = cs_expel_source_sector }

				# Check if there are other planets in the same sector
				if = {
					limit = {
						owner = {
							any_owned_planet = {
								NOT = { is_same_value = event_target:cs_expel_source_planet }
								sector = { is_same_value = event_target:cs_expel_source_sector }
							}
						}
					}
					# Move pops to random planets in same sector
					every_owned_pop_group = {
						save_event_target_as = cs_expel_pop_group
						owner = {
							random_owned_planet = {
								limit = {
									NOT = { is_same_value = event_target:cs_expel_source_planet}
									sector = { is_same_value = event_target:cs_expel_source_sector }
								}
								resettle_pop_group = {
									POP_GROUP = event_target:cs_expel_pop_group
									PLANET = this
									PERCENTAGE = 100
								}
							}
						}
					}
				}
				else = {
					# No other planets in this sector — send to a random other planet, doesn't matter where
					every_owned_pop_group = {
						save_event_target_as = cs_expel_pop_group
						owner = {
								random_owned_planet = {
									resettle_pop_group = {
										POP_GROUP = event_target:cs_expel_pop_group
										PLANET = this
										PERCENTAGE = 100
									}
								}
						}
					}
				}
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

# Toggle decision: Enable Rapid Construction Mode
# Increases building speed at the cost of increased building costs.
# Scales with engineering tech level. Cannot be taken while buildings are under construction.
cs_decision_rapid_construction_enable = {
	icon = decision_resources
	sound = event_administrative_work

	enactment_time = 0

	resources = {
		category = decisions
		cost = {
			unity = 50
		}
	}

	potential = {
		owner = { is_ai = no }
		NOT = { has_planet_flag = cs_rapid_construction_active }
	}

	owned_planets_only = yes

	allow = {
		custom_tooltip = {
			fail_text = "cs_decision_rapid_construction_no_buildings_under_construction"
			NOT = { has_building_construction = yes }
		}
	}

	effect = {
		set_planet_flag = cs_rapid_construction_active
		if = {
			limit = { owner = { has_technology = tech_mega_engineering } }
			add_modifier = { modifier = cs_rapid_construction_t4 days = -1 }
		}
		else_if = {
			limit = { owner = { has_technology = tech_construction_templates } }
			add_modifier = { modifier = cs_rapid_construction_t3 days = -1 }
		}
		else_if = {
			limit = { owner = { has_technology = tech_assembly_pattern } }
			add_modifier = { modifier = cs_rapid_construction_t2 days = -1 }
		}
		else = {
			add_modifier = { modifier = cs_rapid_construction_t1 days = -1 }
		}
	}

	ai_weight = {
		weight = 0
	}
}

# Toggle decision: Disable Rapid Construction Mode
cs_decision_rapid_construction_disable = {
	icon = decision_resources
	sound = event_administrative_work

	enactment_time = 0

	potential = {
		owner = { is_ai = no }
		has_planet_flag = cs_rapid_construction_active
	}

	owned_planets_only = yes

	effect = {
		remove_planet_flag = cs_rapid_construction_active

		if = {
			limit = { has_modifier = cs_rapid_construction_t1 }
			remove_modifier = cs_rapid_construction_t1
		}
		if = {
			limit = { has_modifier = cs_rapid_construction_t2 }
			remove_modifier = cs_rapid_construction_t2
		}
		if = {
			limit = { has_modifier = cs_rapid_construction_t3 }
			remove_modifier = cs_rapid_construction_t3
		}
		if = {
			limit = { has_modifier = cs_rapid_construction_t4 }
			remove_modifier = cs_rapid_construction_t4
		}
	}

	ai_weight = {
		weight = 0
	}
}
